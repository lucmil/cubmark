97 template<int DIM>
98 double RuleChain<DIM>::compute()
99 {
100  ActionDispatch<DIM> dispatch(*this);
101  typename BinaryAdapter<ActionDispatch<2> >::Type* dispatchAdapter(adapter(dispatch));
102  AdapterLock lock(dispatchAdapter);
103 
104  for(size_t i = 0; i < DIM; ++i)
105  {
106  rules[i].setIntegrand(dispatchAdapter->getPtr());
107  }
108 
109  rulePos = 0u;
110  callCnt = 0u;
111  const double result = rules[0].compute();
112 
113  CHECK(rulePos == 0, std::logic_error(
114  "Chain returned before all rules were evaluated."));
115 
116  return result;
117 }
